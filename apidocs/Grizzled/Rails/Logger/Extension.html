def exception(message, ex, progname = nil)
  ex_message = "#{ex.class} (backtrace):\n#{ex.backtrace.join("\n")}"
  if message.nil? || (message.length == 0)
    message = "#{ex_message}"
  else
    message << "\n#{ex_message}"
  end

  do_add(ERROR, message, progname, :flatten => false)
end
